Protocole d'application

Serialisation : Objet

Toutes les commandes pour communiquer avec le serveur seront de la forme :
	commande(arguments);
	elle entrainerons la sérialisations d'objets de type message pour envoyer la commande au serveur.
	
Le serveur répondra toujours avec un objet message sérializé qui contiendra éventuellement un objet personne (ou une liste).

returnmessages : 
    "OK"
    "NicknameAlreadyExists"
    "NameAlreadyExists"
    "NameDoesntExist"
    "NicknameDoesntExist"


Ajouter
    Client => Serveur
        add(nom, surnom);		//ajoute nouvelle personne (avec surnom)
		add(surnomExiste, surnom);	//ajoute un surnom à une personne qui existe
    Serveur => Client
		returnmessage : "OK", "NicknameAlreadyExists", "NameAlreadyExists"

Lister
    Client => Serveur
       list();
    Serveur => Client
       returnmessages : "OK" + liste d'objet personnes

Lister Surnom
    Client => Serveur
       listNickname();
    Serveur => Client
       returnmessages : "OK" + liste d'objet personne

Obtenir Nom
    Client => Serveur
       getName(surnom);
	Serveur => Client
       returnmessages : "OK", "NicknameDoesntExist" + objet personne

Obtenir Surnom
    Client => Serveur
		getNicknames(nom);
    Serveur => Client
       returnmessages : "OK", "NameDoesntExist" + objet personne
	
Supprimer Nom
    Client => Serveur 
       deleteName(personne);
    Serveur => Client
       returnmessages : "OK", "NameDoesntExist"

Supprimer Surnom
    Client => Serveur 
       deleteNickname(surnom);
    Serveur => Client
       returnmessages : "OK", "NicknameDoesntExist"

Modifier Nom
    Client => Serveur
       editName(personne, newNom);
    Serveur => Client
       returnmessages : "OK", "NameAlreadyExists", "NameDoesntExist"

Modifier Surnom
    Client => Serveur
       editNickname(oldSurnom, newSurnom);
    Serveur => Client
       returnmessages : "OK", "NicknameAlreadyExists", "NicknameDoesntExist"

Fin
    Client => Serveur
       exit();
    Serveur => Client
       returnmessages : "OK"
